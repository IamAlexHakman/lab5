#include <stdio.h>
#include <time.h>
#include <stdlib.h>

int main(void){
const int rowCount = 3, colCount = 4; 
int a[rowCount][colCount];
int b[999];
int sum=0;
for (int i = 0; i < rowCount; i++)
for (int j = 0; j < colCount; j++) 
{ 
printf("a[%d][%d]=", i, j);
scanf("%d", &a[i][j]); 
}
for (int i = 0; i < rowCount; i++){
for (int j = 0; j < colCount; j++)
{
printf("%d\t", a[i][j]);
} printf("\n");
}
for (int i = 0; i < rowCount; i++)
for (int j = 0; j < colCount; j++) 
{ 
if(a[i][j]>b[i])
{
  b[i]=a[i][j];
}
}
for (int i = 0; i < rowCount; i++) 
{
  sum+=b[i];
}
printf("сума найбільших елементів її рядків:%d\t", sum);
  return 0;
}
