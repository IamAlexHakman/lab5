#include <stdio.h>
#include <time.h>
#include <stdlib.h>

for (int i=0; i<colCount; i++)
a[i]=Low+rand()%High;
for (int i=0; i<colCount; i++)
if(a[i]>5)
{
sum+=1;
}
for (int i=0; i<colCount; i++) 
printf("a[%d]=%d\n", i, a[i]);
printf("\nкількість елементів масиву, більших за число п’ять:%d\n",sum);
int max=a[0];
for(int i = 0; i < colCount; i++)
 if(a[i] > max)
 max = a[i];
for(int i = 0; i < colCount; i++)
 if(a[i] > max)
 max = a[i];
 for(int i = 0; i < colCount; i++)
{
 if(max==a[i])
 {
 max_el = i;
 }
 }
for(int i = 0; i < colCount; i++)
{
 if(i>max_el)

 }
 }

  return 0;
}
